184:      * Calculates the maximum possible score.
185:      *
186:      * @return integer The maximum possible score.
187:      */
188:     private function calculateMaximumScore()
189:     {
190:         $maxQuery = 0;
191:         $maxTarget = 0;
192: 
193:         for ($i = 0; $i < $this->query->getLength(); ++$i) {
194:             $compound = \substr($this->query, $i, 1);
195:             $maxQuery += $this->substitutionMatrix->getValue(
196:                 $compound, $compound
197:             );
198:         }
199: 
200:         for ($i = 0; $i < $this->target->getLength(); ++$i) {
201:             $compound = \substr($this->target, $i, 1);
202:             $maxTarget += $this->substitutionMatrix->getValue(
203:                 $compound, $compound
204:             );
205:         }

249:     /**
250:      * Calculates the current Cell in the score matrix.
251:      *
252:      * @param Cell $currentCell   The current Cell.
253:      * @param Cell $cellAbove     The Cell above the current Cell.
254:      * @param Cell $cellToLeft    The Cell to the left of the current Cell.
255:      * @param Cell $cellAboveLeft The cell to above and to the left of the
256:      *                            current Cell.
257:      *
258:      * @return void
259:      */
260:     private function calculateCell(
261:         Cell $currentCell, Cell $cellAbove,
262:         Cell $cellToLeft, Cell $cellAboveLeft
263:     ) {
264:         $a = $this->returnCompoundFromQuery($currentCell);
265:         $b = $this->returnCompoundFromTarget($currentCell);
266: 
267:         // Calculate the weight.
268:         $weight = $this->weight($a, $b);
269: 
270:         // Match/Mismatch
271:         $cellAboveLeftScore = $cellAboveLeft->getScore() + $weight;
272:         // Deletion
273:         $cellAboveScore = $cellAbove->getScore() + $weight;
274:         // Insertion
275:         $cellToLeftScore = $cellToLeft->getScore() + $weight;
276: 
277:         if ($cellAboveScore >= $cellToLeftScore) {
278:             if ($cellAboveLeftScore >= $cellAboveScore) {
279:                 if ($cellAboveLeftScore > self::INITIAL_SCORE) {
280:                     $currentCell->setScore($cellAboveLeftScore);
281:                     $currentCell->setPreviousCell($cellAboveLeft);
282:                 }
283:             } else {
284:                 if ($cellAboveScore > self::INITIAL_SCORE) {
285:                     $currentCell->setScore($cellAboveScore);
286:                     $currentCell->setPreviousCell($cellAbove);
287:                 }
288:             }
289:         } else {
290:             if ($cellAboveLeftScore >= $cellToLeftScore) {
291:                 if ($cellAboveLeftScore > self::INITIAL_SCORE) {
292:                     $currentCell->setScore($cellAboveLeftScore);
293:                     $currentCell->setPreviousCell($cellAboveLeft);
294:                 }
295:             } else {
296:                 if ($cellToLeftScore > self::INITIAL_SCORE) {
297:                     $currentCell->setScore($cellToLeftScore);
298:                     $currentCell->setPreviousCell($cellToLeft);
299:                 }
300:             }
301:         }
302: 
303:         if ($currentCell->getScore() > $this->scoreCell->getScore()) {
304:             $this->scoreCell = $currentCell;
305:         }
306:     }

352:     private function traceback()
353:     {
354:         // Start at the Cell with the score.
355:         $alignedQuery = $this->returnCompoundFromQuery($this->scoreCell);
356:         $alignedTarget = $this->returnCompoundFromTarget($this->scoreCell);
357:         $previousCell = $this->scoreCell;
358:         $currentCell = $this->scoreCell->getPreviousCell();
359: 
360:         while (true === $this->isTracebackNotDone($currentCell)) {
361:             $rowDiff = ($previousCell->getRow() - $currentCell->getRow());
362:             if (1 === $rowDiff) {
363:                 $alignedTarget = $this->returnCompoundFromTarget($currentCell)
364:                     . $alignedTarget;
365:             } else {
366:                 $alignedTarget = '-' . $alignedTarget;
367:             }
368: 
369:             $columnDiff = ($previousCell->getColumn()
370:                 - $currentCell->getColumn());
371:             if (1 === $columnDiff) {
372:                 $alignedQuery = $this->returnCompoundFromQuery($currentCell)
373:                     . $alignedQuery;
374:             } else {
375:                 $alignedQuery = '-' . $alignedQuery;
376:             }
377: 
378:             $previousCell = $currentCell;
379:             $currentCell = $currentCell->getPreviousCell();
380:         }
